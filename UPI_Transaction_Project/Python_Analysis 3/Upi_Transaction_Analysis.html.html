# T-test: Compare transaction amounts 
# between mobile and tablet

mobile = upi_transaction_history[
    upi_transaction_history['device_type']
    =='android']['amount']

tablet = upi_transaction_history[
    upi_transaction_history['device_type']
    =='tablet']['amount']

t_stat, p_value = stats.ttest_ind(mobile, tablet)

print("=== T-Test Results ===")
print(f"T-Statistic: {t_stat:.4f}")
print(f"P-Value: {p_value:.4f}")

if p_value < 0.05:
    print("Result: SIGNIFICANT difference between")
    print("android and tablet transaction amounts")
else:
    print("Result: NO significant difference")

# Visualize
plt.figure(figsize=(10, 6))
sns.boxplot(data=upi_transaction_history[
    upi_transaction_history['device_type'].isin(
    ['android', 'tablet'])],
    x='device_type', y='amount',
    palette='Set2')
plt.title('T-Test: Amount by Device Type')

plt.show()

# ANOVA: Transaction amounts across 
# all device types

groups = []
device_types = upi_transaction_history[
    'device_type'].unique()

for device in device_types:
    group = upi_transaction_history[
        upi_transaction_history['device_type']
        ==device]['amount']
    groups.append(group)

f_stat, p_value = stats.f_oneway(*groups)

print("=== ANOVA Results ===")
print(f"F-Statistic: {f_stat:.4f}")
print(f"P-Value: {p_value:.4f}")

if p_value < 0.05:
    print("Result: SIGNIFICANT difference in amounts")
    print("across device types")

# Chi-square: Association between 
# fraud_flag and channel

contingency_table = pd.crosstab(
    upi_transaction_history['fraud_flag'],
    upi_transaction_history['channel'])

print("=== Contingency Table ===")
print(contingency_table)

chi2, p_value, dof, expected = \
    stats.chi2_contingency(contingency_table)

print("\n=== Chi-Square Test Results ===")
print(f"Chi-Square Statistic: {chi2:.4f}")
print(f"P-Value: {p_value:.4f}")
print(f"Degrees of Freedom: {dof}")

if p_value < 0.05:
    print("Result: SIGNIFICANT association between")
    print("fraud flag and channel")

    print("Result: NO significant association")
